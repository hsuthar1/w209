def q3():
    # gets counts per day
    temp = df.groupby(pd.DatetimeIndex(pd.to_datetime(df["Timestamp"])).date).count()
#     print(temp)
    
    # builds the x domain for our bar chart
    x = range(1,len(temp.index.tolist())+1)
#     print(x)
    
    # because there are so many days, builds a x ticks list that skips over certain days
    xtx = np.arange(min(x), max(x)+1, 30.0)
#     print(xtx)
    
    # and because the x ticks are skipped, have to skip over certain x labels too
    xlabs = temp.index[1::30]
#     print(xlabs)
    
    # builds the y range for our bar chart
    y = temp["car-id"]
#     print(y)
    
    # builds a line chart
    plt.figure(figsize=(24,8))
    plt.plot(x,y)
    plt.title("Count of Daily Visits")
    plt.xticks(xtx, xlabs)
    plt.xlabel("Day")
    plt.ylabel("number of cars")
    plt.show()

def q4():
    # gets counts per day
    temp = {}
    for car_type in df["car-type"].unique(): 
        g = df[df["car-type"] == car_type].groupby(pd.DatetimeIndex(df[df["car-type"] == car_type]["Timestamp"]).date).count()["car-type"]
        temp[car_type] = g
#         print(g.head())
#     print(len(temp))
#     print(temp.keys())
#     print(temp.values())
    
    # builds a line chart
    plt.figure(figsize=(24,8))
    for idx, s in temp.items():
        plt.plot(s, label=idx)
    plt.title("Count of Daily Visits per Car Type")
    plt.xlabel("Day")
    plt.ylabel("number of cars")
    plt.legend()
    plt.show()

def q6():
    # gets counts of daily visits per day of week
    temp = {}
    for car_type in df["car-type"].unique(): 
        g = df[df["car-type"] == car_type].groupby(pd.DatetimeIndex(df[df["car-type"] == car_type]["Timestamp"]).dayofweek).count()["car-type"]
        temp[car_type] = g
#     print(temp)
    
    # for each car type, builds a bar chart
    for idx, s in temp.items():
#         print(s)
        
        # builds the x domain for our bar chart
        x = range(1,len(s.tolist())+2)
#         print(x)
        
        # builds the y range for our bar chart
        y = s.values
        y = np.append(y,[0]) # adds a 0 so that we have room for the legend
#         print(y)
    
        # builds a bar chart
        plt.bar(x, y, label=idx)
        
    plt.title("Number of daily visits per day of week")
    plt.xlabel("cart type")
    plt.ylabel("number of cars")
    plt.legend(loc="right")
    plt.show()
